153 public void connectarAmbCanonada(NodeAixeta node1, NodeAixeta node2, float c) {
154 Node n1 = _xarxa.getNode(node1.id());
155 Node n2 = _xarxa.getNode(node2.id());
156
157 if(n1 == null || n2 == null) {
158 throw new NoSuchElementException("Algun dels nodes no pertany a la xarxa");
159 }
160
161 if(node1 instanceof Terminal) {
162 throw new IllegalArgumentException("El node " + node1.id() + " és un node terminal");
163 }
164
165 //mirar si els dos nodes ja estan connectats
166 if (n1.hasEdgeBetween(n2)) {
167 throw new IllegalArgumentException("Nodes ja connectats");
168 }
169
170 //Capacitat negativa?
171 if (c < 0) {
172 throw new IllegalArgumentException("La capacitat no pot ser negativa");
173 }
174
175 String idCanonada = node1.id() + "-" + node2.id();
176 Edge e = _xarxa.addEdge(idCanonada, node1.id(), node2.id(), true);
177
178 Canonada pipe;
179 if(node2 instanceof Origen) {
180 Connexio conn = new Connexio(node2.id(), node2.coordenades());
181 Node n = _xarxa.getNode(conn.id());
182 n.removeAttribute("aixeta");
183 n.setAttribute("aixeta", conn);
184
185 n.removeAttribute("ui.class");
186 n.setAttribute("ui.class", "connexio");
187
188 pipe = new Canonada(node1, conn, c);
189 }
190 else{
191 pipe = new Canonada(node1, node2, c);
192 }
193
194 e.setAttribute("canonada", pipe);
195 }

290 public void recular(int nPassos) {
291 if(nPassos <= 0)

296 int nombrePassos = nPassos;

298 while (!_modifAixetes.empty() && nombrePassos > 0) {
299 // Desempilar
300 String contingut = _modifAixetes.pop();
301 String[] parts = contingut.split("-");
302
303 // Si aquella modificació ha creat un canvi d'estat, revertir-lo
304 // ex: aixeta oberta i es vol obrir, no hi ha hagut un canvi i, per tant, s'ignora aquest pas
305 if (parts[1].equals("si")) {
306 Node n = _xarxa.getNode(parts[0]);
307 NodeAixeta aixeta = (NodeAixeta) n.getAttribute("aixeta");
308 aixeta.canviarEstat();
309 }

360 public float cabal(NodeAixeta node) {
361 if (_xarxa.getNode(node.id()) == null) {
362 throw new NoSuchElementException("Aquest node no pertany a la xarxa");
363 }
364
365 if(node instanceof Terminal terminal) {
366 if(terminal.demandaActual() == 0) {
367 return 0;
368 }
369 }
370
371 if(node instanceof Origen origen) {
372 Node n = _xarxa.getNode(node.id());
373 if(n.getAttribute("ui.class").equals("origen")) {
374 if (origen.cabal() <= demanda(origen)) {
375 return origen.cabal();
376 } else {
377 return demanda(origen);
378 }
379 }
380 }
381
382 float cabalTotal = 0;
383 Iterator<Canonada> pipes = entrades(node);
384 while (pipes.hasNext()) {
385 Canonada pipe = pipes.next();
386 cabalTotal += calculCabalCanonada(pipe);
387 }
388
389 return cabalTotal;

396 public float calculCabalCanonada(Canonada pipe){
397 float demandaTotalCanonades = 0;
398 Iterator<Canonada> pipes = sortides(pipe.node1());
399 while(pipes.hasNext())
400 {//Suma de les capacitats de les canonades que entren al segon node de pipe
401 Canonada c = pipes.next();
402 demandaTotalCanonades += calculDemandaCanonada(c);
403 }
404
405 float cabalNode = cabal(pipe.node1());
406 //Repartició cabal
407 float demandaCanonada = calculDemandaCanonada(pipe);
408 float cabalCanonada = (demandaCanonada / demandaTotalCanonades) * cabalNode;
409
410 return cabalCanonada;
411 }

449 private float calculDemandaCanonada(Canonada pipe) {
450 float capacitatCanonadesTotal = 0;
451 Iterator<Canonada> pipes = entrades(pipe.node2());
452 while(pipes.hasNext())
453 {//Suma de les capacitats de les canonades que entren al segon node de pipe
454 Canonada c = pipes.next();
455 capacitatCanonadesTotal += c.capacitat();
456 }
457
458 float demandaNode = demanda(pipe.node2());
459 //Repartició demanda
460 float demandaCanonada = (pipe.capacitat() / capacitatCanonadesTotal) * demandaNode;
461
462 if(demandaCanonada > pipe.capacitat()) {
463 return pipe.capacitat();
464 }
465 else {
466 return demandaCanonada;
467 }

556 private void copiaGraf(Node node, Set<Node> visitats, Graph subGraph) {
557 visitats.add(node);
558 Node n = subGraph.addNode(node.getId());
559
560 Node nodeXarxa = _xarxa.getNode(node.getId());
561 NodeAixeta nodeAixeta = (NodeAixeta) nodeXarxa.getAttribute("aixeta");
562
563 if (nodeAixeta instanceof Connexio nodeConnexio) {
564 n.setAttribute("aixeta", nodeConnexio);
565 n.setAttribute("ui.class", "connexio");
566
567 }
568 else if(nodeAixeta instanceof Origen nodeOrigen) {
569 n.setAttribute("aixeta", nodeOrigen);
570 n.setAttribute("ui.class", "origen");
571
572 } else if (nodeAixeta instanceof Terminal nodeTerminal) {
573 n.setAttribute("aixeta", nodeTerminal);
574 n.setAttribute("ui.class", "terminal");
575
576 }
577
578 n.setAttribute("ui.label", node.getId());
579
580 Iterator<Edge> iterator = node.edges().iterator();
581 while (iterator.hasNext()) {
582 Edge edge = iterator.next();
583 Node adjacent = edge.getOpposite(node);
584
585 if (!visitats.contains(adjacent)) {
586 copiaGraf(adjacent, visitats, subGraph);
587 }
588
589 // Verifiquem si els dos extrems de l'aresta estan al subgraf abans d'afegir l'aresta
590 if (subGraph.getNode(node.getId()) != null && subGraph.getNode(adjacent.getId()) != null) {
591 // Afegim l'aresta al subgraf
592 if (subGraph.getEdge(edge.getId()) == null) {
593 Node nodeInicial = edge.getSourceNode();
594 if (nodeInicial.equals(node)) {
595 // Si el node d'origen es el mateix que el node actual, mantenim la mateixa direccio
596 subGraph.addEdge(edge.getId(), node.getId(), adjacent.getId(), true);
597 } else {
598 subGraph.addEdge(edge.getId(), adjacent.getId(), node.getId(), true);
599 }
600
601 if (edge.hasAttribute("canonada")) {
602 subGraph.getEdge(edge.getId()).setAttribute("canonada", edge.getAttribute("canonada"));
603 }
604 }
605 }
606 }
