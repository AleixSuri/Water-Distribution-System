45 private static boolean recorregutCicle(Node node, Set<Node> visitats) {
46 // Si el node actual ja s'ha visitat retorna true, hi ha un cicle
47 if (visitats.contains(node)) {
48 return true;
49 }
50
51 // S'afegeix node actual al conjunt de visitats
52 visitats.add(node);
53
54 // Es recorre per totes les arestes que surten del node
55 Iterator<Edge> iterator = node.leavingEdges().iterator();
56 while (iterator.hasNext()) {
57 // S'obté el node adjacent a l'actual
58 Node adjacent = iterator.next().getTargetNode();
59
60 // Si hi ha cicles retorna true
61 if (recorregutCicle(adjacent, visitats)) {
62 return true;
63 }
64 }
65
66 // Es treu el node actual del conjunt de nodes visitats
67 visitats.remove(node);
68 return false;
69 }

100 {
101 visitats.add(node);

175 public static Set<NodeAixeta> aixetesTancar(Xarxa x, Map<Terminal, Boolean> aiguaArriba)
176 {
177
178 Set<NodeAixeta> tancarAixetes = new HashSet<>();
179 //Buscar aixetes trencades
180 for (Map.Entry<Terminal, Boolean> ItAiguaArriba : aiguaArriba.entrySet()) {
181 Terminal terminal = ItAiguaArriba.getKey();
182 Boolean aiguaArribaValor = ItAiguaArriba.getValue();
183
184 if(!aiguaArribaValor && terminal.aixetaOberta()) {//si no arriba aigua
185 Iterator<Canonada> entradesT = x.entrades(terminal);
186 while (entradesT.hasNext()) { //mirem aixetes superiors
187 tancarAixetes.add(entradesT.next().node1()); //guardem aixeta trencada
188 }
189 }
190
191 }
192 //buscar nodes fills que no estan trencats
193 Set<NodeAixeta> tancarAixetesFills = new HashSet<>();//set que serveix per saber els nodes fills
194 Iterator<NodeAixeta> iterator = tancarAixetes.iterator();
195 while (iterator.hasNext()) {
196 NodeAixeta aixeta = iterator.next();
197 eliminarFills(tancarAixetes, aixeta, x,tancarAixetesFills);
198 }
199 //treiem els nodes fills
200 tancarAixetes.removeAll(tancarAixetesFills);
201 return tancarAixetes;

209 if(aixeta instanceof Connexio) {

215 if(nodoNext instanceof Connexio) {
216 if (tancarAixetes.contains(nodoNext)) {
217 tancarAixetesFills.add(nodoNext);
218 }
219 if (!tancarAixetes.isEmpty()) {
220 eliminarFills(tancarAixetes, nodoNext, x,tancarAixetesFills);
221 } else {
222 acabar = true;
223 }
224 }

240
241 // Itera sobre la llista ordenada per trobar posició pel nou node

243 while (iterador.hasNext() && !trobat) {
244 NodeAixeta node = iterador.next();
245 if ((distancia < c.distancia(node.coordenades())) || (distancia == c.distancia(node.coordenades()) && nodeSet.idPetit(node))) {
246 //Afegir el node
247 iterador.previous();
248 iterador.add(nodeSet);
249 trobat = true; //Sortir del while
250 }

252 }
253 if (!trobat) {
254 llistaOrdenats.add(nodeSet);
255 }

431 private static float buscarCabalMaxim(NodeAixeta nodeAixeta,float cabalPossible, Graph graph) {
432 float cabalTotal = 0;
433 if(nodeAixeta instanceof Terminal terminal){
434 cabalTotal = cabalPossible;
435 }
436 else{
437 //per tots els edges sortida del node
438 Node nodeIterador = graph.getNode(nodeAixeta.id());
439 Iterator<Edge> edgeIterator = nodeIterador.leavingEdges().iterator();
440 while (edgeIterator.hasNext()) {
441 Edge aresta = edgeIterator.next();
442 Canonada canonada = (Canonada) aresta.getAttribute("canonada");
443 //comprovar si cabalPossible és més petit i cridar funció recursiva
444 float cabalCanonada = 0;
445 if (cabalPossible > canonada.capacitat()) {
446 cabalCanonada = buscarCabalMaxim(canonada.node2(), canonada.capacitat(), graph);
447 }
448 else cabalCanonada = buscarCabalMaxim(canonada.node2(), cabalPossible, graph);
449
450 cabalTotal += cabalCanonada;
451 cabalPossible -= cabalCanonada;
452 //guardar cabal a dibuix
453 aresta.setAttribute("ui.label",cabalCanonada + "/" + canonada.capacitat());
454
455 }
456 }
457 return cabalTotal;
